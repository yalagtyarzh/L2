```go
package main

import (
	"fmt"
	"math/rand"
	"time"
)

func asChan(vs ...int) <-chan int {
	c := make(chan int)

	go func() {
		for _, v := range vs {
			c <- v
			time.Sleep(time.Duration(rand.Intn(1000)) * time.Millisecond)
		}

		close(c)
	}()
	return c
}

func merge(a, b <-chan int) <-chan int {
	c := make(chan int)
	go func() {
		for {
			select {
			case v := <-a:
				c <- v
			case v := <-b:
				c <- v
			}
		}
	}()
	return c
}

func main() {

	a := asChan(1, 3, 5, 7)
	b := asChan(2, 4, 6, 8)
	c := merge(a, b)
	for v := range c {
		fmt.Println(v)
	}
}

```

Ответ:
```
В начале выводятся значения 1, 2, 3, 4, 5, 6, 7, 8, а позже будут бесконечно выводиться одни нули. Такое происходит,
поскольку мы продолжаем читать данные из уже закрытого канала и в конструкции select мы всегда получаем нули. Чтобы
такого избежать, можно воспользоваться механикой чтения канала. Дело в том, что при чтении канала возвращаются два
значения - первым является само значение из канала, а второе булево, указывающее, является ли данный канал закрытым.
С помощью данной механики с помощью второй переменной можно провалидировать открытость канала и ввести дополнительную
логику.
```
